# 最长连续序列
这题不难想到用哈希，关键是其中两个优化，一个是从哈希表中去遍历，一个是判断不是序列起点就跳过
```C++
int longestConsecutive(vector<int>& nums) {
    if (nums.empty()) {
        return 0;
    }
    std::unordered_set<int32_t> record;
    int32_t result = 1;
    for (int32_t i = 0; i < nums.size(); ++i) {
        record.emplace(nums[i]);
    }
    // 从已经去重的数组中遍历
    for (auto num : record) {
        // 不是起点就跳过
        if (record.count(num - 1)) {
            continue;	
        }
        int32_t len = 1;
        while (record.count(num + 1) != 0) {
            len++;
            num = num + 1;
        }
        result = std::max(result, len);
    }
    return result;
}
```
